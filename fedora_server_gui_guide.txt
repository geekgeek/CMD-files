#GUI
dnf -y group install "KDE Plasma Workspaces" 

echo "exec /usr/bin/startkde" >> ~/.xinitrc 

systemctl set-default graphical.target

reboot

#XRDP
dnf -y install xrdp 

systemctl start xrdp 

systemctl enable xrdp 

#firewall
firewall-cmd --add-port=3389/tcp --permanent 
firewall-cmd --reload
